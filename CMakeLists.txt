#define a macro to simplify adding tests
macro (do_test arg result)
  add_test(${arg}.test ${PROJECT_BINARY_DIR}/TESTS/${arg}/autotest.sh ${arg})
  set_tests_properties (${arg}.test
    PROPERTIES PASS_REGULAR_EXPRESSION ${result}
  )
endmacro (do_test)
